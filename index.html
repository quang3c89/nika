📋 SQL tạo bảng (chạy trong Supabase SQL Editor)
create table if not exists players (id text primary key,name text not null,phone text,rank text,unit text,avatar text,status text default 'pending',tournament_ids jsonb default '[]',created_at timestamptz default now()); create table if not exists tournaments (id text primary key,name text,format text,description text,player_ids jsonb default '[]',matches jsonb default '[]',playin_matches jsonb default '[]',status text default 'draft',main_size int,created_at timestamptz default now()); create table if not exists events (id text primary key,name text,prize_text text,prize_image text,winner_id text,created_at timestamptz default now()); create table if not exists participants (id text primary key,event_id text,name text,order_number int,lucky_number text,created_at timestamptz default now()); create table if not exists logs (id text primary key,action text,icon text,category text,timestamp bigint,created_at timestamptz default now()); create table if not exists settings (id text primary key,brand_name text,sub_text text,logo_data text,display_font text,body_font text,font_size int,colors jsonb,ui_texts jsonb,admin_email text,created_at timestamptz default now());
